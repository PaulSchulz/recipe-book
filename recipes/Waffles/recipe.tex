\recette{Waffles}
\preptime{5 min} \cooktime{40 min} \cake{8} \bobcat{Waffle Maker}

\recipe{
  \unit[1]{cup}   & Self-raising flour \\
  1               & Egg \\ 
  \unit[1/4]{cup} & Oil \\
  \unit[3/4]{cup} & Milk \\
  \unit[1]{tbsp}  & Sugar \\
  \unit[1/2]{tsp} & Vanilla essence \\
}{
\item Mix flour, egg, oil, milk, sugar and vanilla essence in a bowl until smooth, without lumps.

\item Allow waffler maker to heat up.

\item Pour \unit[1/4]{cup} (approx.) of batter into waffle maker, close and allow to cook for 4-5 minutes.

\item Remove when waffle is nicely browned.

\item Repeat with rest of batter.

}

\info{Top with jelly and cream, or icecream.}
